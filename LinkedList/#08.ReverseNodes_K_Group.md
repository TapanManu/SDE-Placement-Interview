# #8. Reverse Nodes in k-Group

Given a linked list, reverse the nodes of a linked list k at a time and return its modified list.


```
Given this linked list: 1->2->3->4->5

For k = 2, you should return: 2->1->4->3->5
For k = 3, you should return: 3->2->1->4->5
```

**Python**
```python
def reverseKGroup(self, head, k):
    dummy = jump = ListNode(0)
    dummy.next = l = r = head
    
    while True:
        count = 0
        while r and count < k:
            r = r.next
            count += 1
        if count == k:
            prev, cur = r, l
            for _ in range(k):
                cur.next, cur, prev = prev, cur.next, cur
            jump.next, jump, l = prev, l, r
        else:
            break
    
    return dummy.next
```

```python
def reverseKGroup(self, head, k):
    def reverse(head, prev):
        q = k
        cur = head
        while cur and q:
            cur.next, prev, cur = prev, cur, cur.next
            q -= 1
        return prev, head, cur
    
    prev = start = main_head = None
    cur_next = head
    bool_head = True
    
    def check(cur_next):
        p = 0
        while cur_next and p < k:
            cur_next = cur_next.next
            p += 1
        return False if p < k else True
    
    while check(cur_next):
        old_temp = start
        prev, start, cur_next = reverse(cur_next, None)
        if old_temp is not None:
            old_temp.next = prev
        if bool_head:
            main_head = prev
            bool_head = False
    
    if start is None:
        main_head = cur_next
    elif cur_next is not None:
        start.next = cur_next
    
    return main_head
```
