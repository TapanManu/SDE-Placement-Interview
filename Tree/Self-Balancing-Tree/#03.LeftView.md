## #3. Left View

```
	   7
	 /	\
	2    11
 / \  / \
4   3 8  15

Ouput: 7 2 4 3
```

```cpp
vector<int> leftView(Node *root) {
	if (root == NULL) return {};

	queue<Node*> q;
	q.push(root);
	q.push(NULL);

	vector<int> result;
	int first_ele = root->key;
	bool first = true;

	while (q.size() > 1) {
		Node* cur = q.front();
		q.pop();

		if (first) {
			result.push_back(cur->key);
			first = false;
		}

		if (cur == NULL) {
			q.push(NULL);
			first = true;
			continue;
		}

		if (cur->left != NULL) q.push(cur->left);
		if (cur->right != NULL) q.push(cur->right);
	}

	return result;
}
```
