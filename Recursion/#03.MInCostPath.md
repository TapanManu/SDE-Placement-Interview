### #3. Min Cost Path

```
int cost[R][C] = 
    {{1, 2, 2},
		{3, 1, 5},
		{1, 5, 3}, 
    {3, 2, 8}
	};
```

<img src="https://www.geeksforgeeks.org/wp-content/uploads/dp2.png">

```cpp
#define C 3
#define R 4

int minCost(int cost[R][C], int m, int n) {
	if (m < 0 || n < 0)
		return INT_MAX;

	if (m == 0 && n == 0)
		return cost[m][n];

	return cost[m][n] + min(min(minCost(cost, m - 1, n), minCost(cost, m, n - 1)),
	                        minCost(cost, m - 1, n - 1));
}
```
